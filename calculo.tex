\frame{\frametitle{Derivadas Parciais}


$$\left( \frac{\partial f}{\partial x} \right)_y =
\lim_{h\to 0} \frac {f(x+h,y)-f(x,y)}{h}$$

$$\left( \frac{\partial f}{\partial y} \right)_x =
\lim_{h\to 0} \frac {f(x,y+h)-f(x,y)}{h}$$


\begin{center}
  \includegraphics[scale=0.5]{partial}
\end{center}
}

\frame{\frametitle{Diferencial Total}

$$df= \left( \frac{\partial f}{\partial x} \right)_y dx+\left( \frac{\partial f}{\partial y} \right)_x dy$$

}

\frame{\frametitle{Extremos de Funções}
$$df=0$$

$$\left( \frac{\partial f}{\partial x} \right)_y=0$$

$$\left( \frac{\partial f}{\partial y} \right)_x=0$$
}

\frame{\frametitle{Extremos de Funções com Vínculos}

 $f=f(x,y)$ sujeita ao vínculo $g(x,y)=0$. 

Multiplicador de Lagrange:

Minimize $h(x,y)=f(x,y)-\lambda g(x,y)$ e resolva as equações:

$$\left( \frac{\partial h}{\partial x} \right)_y=0$$

$$\left( \frac{\partial h}{\partial y} \right)_x=0$$

$$g(x,y)=0$$

}
